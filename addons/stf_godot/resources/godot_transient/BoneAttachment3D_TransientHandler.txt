class_name BoneAttachment3D_TransientHandler
extends STF_Handler
## Export the children of a BoneAttachment3D
##
## [url]https://docs.stfform.at/resources/stf/stf_node.html[/url]

func _get_stf_type() -> String: return ""
func _get_priority() -> int: return 10000
func _get_stf_category() -> String: return "transient"
func _get_like_types() -> Array[String]: return []
func _get_godot_types() -> Array[String]: return ["BoneAttachment3D"]
func _is_transient() -> bool: return true

func _check_godot_object(godot_object: Variant) -> int:
	return 10000 if godot_object is BoneAttachment3D else -1

func _import(context: STF_ImportContext, stf_id: String, json_resource: Dictionary, context_object: Variant, instance_context: Variant) -> ImportResult:
	return null

func _export(context: STF_ExportContext, godot_object: Variant, context_object: Variant, instance_context: Variant) -> ExportResult:
	var godot_node: BoneAttachment3D = godot_object
	var children = []
	for child in godot_node.get_children():
		if(child is not BoneAttachment3D):
			continue
		var child_id = context.export(child, "node", godot_node, instance_context)
		if(child_id >= 0):
			children.append(child_id)

	return null
